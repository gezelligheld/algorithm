默认记录一个最小值，如果第 i 个值比最小的值还要小，则交换位置

```javascript
function selectSort(arr) {
  for (let i = 0; i < arr.length; i++) {
    let min = i;
    // i相当于已排序元素的末尾索引，这里从未排序的元素中找最小值
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }

    let t = arr[i];
    arr[i] = arr[min];
    arr[min] = t;
  }
  return arr;
}
```
