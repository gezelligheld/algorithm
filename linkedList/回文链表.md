判断一个链表是否为回文链表

示例
输入: 1->2->2->1
输出: true

方法一：将链表的值放到数组中，遍历数组判断是否为回文

```js
function isPalindrome(head) {
    const vals = [];
    let cur = head;
    while (cur) {
        vals.push(cur.val);
        cur = cur.next;
    }

    for (let i = 0, j = vals.length - 1; i < vals.length; i ++, j --) {
        if (vals[i] !== vals[j]) {
            return false;
        }
    }

    return true;
};
```

方法二：找到链表的后半段，将其反转，然后与前半段进行比较

```js
function isPalindrome(head) {
    // 获取链表的后一半
    let slow = head;
    let fast = head;
    while (fast.next && fast.next.next) {
        fast = fast.next.next;
        slow = slow.next;
    }

    // 反转
    let cur = slow.next;
    let pre = null;
    while (cur) {
        const node = cur.next;
        cur.next = pre;
        pre = cur;
        cur = node;
    }

    let p1 = head;
    let p2 = pre;
    while (p1 && p2) {
        if (p1.val !== p2.val) {
            return false;
        }
        p1 = p1.next;
        p2 = p2.next;
    }
    return true;
}
```
